function loginStudent(studentId){
  const id=clean_(studentId);
  if(!id)return {success:false,message:'Please enter your student ID.'};
  const s=findStudent_(id);
  if(!s)return {success:false,message:'Student ID not found.'};
  if(!isTrue_(s.active))return {success:false,message:'This student account is not active.'};
  ensureProfile_(s);
  return {success:true,student:s,dashboard:getStudentDashboard(id)};
}

function loginStaff(role,name,password){
  role=clean_(role).toLowerCase();
  name=clean_(name);
  password=clean_(password);
  const tab=role==='administrator'?TAB.ADMINS:TAB.TEACHERS;
  const rows=objects_(tab);
  const nameKey=role==='administrator'?'Administrator':'Teacher';
  const match=rows.find(r=>clean_(r[nameKey]).toLowerCase()===name.toLowerCase());
  if(!match || clean_(match.Password)!==password || !isTrue_(match.Active)){
    return {success:false,message:'Account name or password is incorrect.'};
  }
  return {success:true,user:{role,name:match[nameKey]},dashboard:getStaffDashboard(role,match[nameKey])};
}

function savePortfolioArtifact(a){
  const s=findStudent_(a.studentId);

  const displayName=clean_(a.publicDisplayName)||defaultDisplayName_(s);
  const artifactId=id_('ARTIFACT');

  let driveFileId='',driveUrl='',thumbnailUrl='',fileName='',fileType='';
  if(a.fileData && a.fileName){
    const saved=savePortfolioFile_(s,a);
    driveFileId=saved.fileId;
    driveUrl=saved.url;
    thumbnailUrl=saved.thumbnailUrl;
    fileName=saved.fileName;
    fileType=saved.fileType;
  }

  appendObject_(TAB.PORTFOLIO,{
    'Artifact ID':artifactId,
    'Student ID':s.studentId,
    'Student Name':s.fullName,
    'Public Display Name':displayName,
    'Grade':s.grade,
    'Teacher':s.teacher,
    'Project ID':clean_(a.projectId),
    'Title':clean_(a.title)||'Untitled Portfolio Item',
    'Description':clean_(a.description),
    'Reflection':clean_(a.reflection),
    'File Name':fileName,
    'File Type':fileType,
    'Drive File ID':driveFileId,
    'Drive URL':driveUrl,
    'Thumbnail URL':thumbnailUrl,
    'Visibility':visibility,
    'Proud Of This':a.proudOfThis===true||clean_(a.proudOfThis).toLowerCase()==='true',
    'Status':visibility==='Submit to Compass Showcase'?'Pending Review':'Saved',
    'Submitted Date':new Date(),
    'Approved By':'',
    'Approved Date':''
  });

  return {success:true,message:visibility==='Submit to Compass Showcase'?'Saved and submitted for review.':'Saved to your portfolio.',artifactId};
}

function savePortfolioFile_(student,a){
  const root=getOrCreateFolder_('Compass Portfolio');
  const gradeFolder=getOrCreateChildFolder_(root,'Grade '+student.grade);
  const teacherFolder=getOrCreateChildFolder_(gradeFolder,clean_(student.teacher)||'Unassigned');
  const studentFolder=getOrCreateChildFolder_(teacherFolder,student.studentId+' - '+student.fullName);

  const raw=String(a.fileData);
  const match=raw.match(/^data:(.+?);base64,(.+)$/);
  if(!match)throw new Error('The uploaded file could not be read.');

  const mime=match[1];
  const bytes=Utilities.base64Decode(match[2]);
  const safeName=String(a.fileName).replace(/[^\w.\- ()]/g,'_');
  const blob=Utilities.newBlob(bytes,mime,safeName);
  const file=studentFolder.createFile(blob);

  file.setSharing(DriveApp.Access.ANYONE_WITH_LINK,DriveApp.Permission.VIEW);

  return {
    fileId:file.getId(),
    url:file.getUrl(),
    thumbnailUrl:'https://drive.google.com/thumbnail?id='+file.getId()+'&sz=w1000',
    fileName:safeName,
    fileType:mime
  };
}

function getOrCreateFolder_(name){
  const folders=DriveApp.getFoldersByName(name);
  return folders.hasNext()?folders.next():DriveApp.createFolder(name);
}

function getOrCreateChildFolder_(parent,name){
  const folders=parent.getFoldersByName(name);
  return folders.hasNext()?folders.next():parent.createFolder(name);
}

function approveShowcaseArtifact(artifactId,approvedBy,approved){
  const artifact=findBy_(TAB.PORTFOLIO,'Artifact ID',artifactId);
  if(!artifact)throw new Error('Portfolio item not found.');

  if(approved!==true){
    updateField_(TAB.PORTFOLIO,'Artifact ID',artifactId,{
      'Status':'Returned to Student',
      'Approved By':clean_(approvedBy),
      'Approved Date':new Date()
    });
    return {success:true,message:'The item was returned to the student.'};
  }

  const showcaseId=id_('SHOWCASE');
  appendObject_(TAB.SHOWCASE,{
    'Showcase ID':showcaseId,
    'Artifact ID':artifactId,
    'Student ID':artifact['Student ID'],
    'Public Display Name':artifact['Public Display Name']||privacyNameFromFull_(artifact['Student Name']),
    'Grade':artifact.Grade,
    'Project Title':artifact.Title,
    'Subject Area':'Portfolio',
    'Description':artifact.Description,
    'Reflection':artifact.Reflection,
    'Drive URL':artifact['Drive URL'],
    'Thumbnail URL':artifact['Thumbnail URL'],
    'Approved':true,
    'Approved By':clean_(approvedBy),
    'Approved Date':new Date()
  });

  updateField_(TAB.PORTFOLIO,'Artifact ID',artifactId,{
    'Status':'Published to Showcase',
    'Approved By':clean_(approvedBy),
    'Approved Date':new Date()
  });

  return {success:true,message:'Published to the Compass Showcase.'};
}

function getApprovedShowcase_(){
  return objects_(TAB.SHOWCASE).filter(r=>isTrue_(r.Approved)).reverse();
}

function defaultDisplayName_(s){
  return s.firstName+' '+(s.lastName?String(s.lastName).charAt(0)+'.':'');
}

function privacyNameFromFull_(fullName){
  const parts=clean_(fullName).split(/\s+/).filter(Boolean);
  if(!parts.length)return 'Compass Student';
  return parts[0]+(parts.length>1?' '+parts[parts.length-1].charAt(0)+'.':'');
}

function ensureHeaders_(name,headers){
  let sh=SS.getSheetByName(name)||SS.insertSheet(name);
  let existing=sh.getLastColumn()?sh.getRange(1,1,1,sh.getLastColumn()).getDisplayValues()[0]:[];
  headers.forEach(h=>{if(!existing.includes(h)){sh.getRange(1,sh.getLastColumn()+1).setValue(h);existing.push(h)}});
  sh.getRange(1,1,1,sh.getLastColumn()).setFontWeight('bold').setBackground('#0b1f3a').setFontColor('#fff');
  sh.setFrozenRows(1);
}

function objects_(tab){
  const sh=sheet_(tab),vals=sh.getDataRange().getDisplayValues();
  if(vals.length<2)return [];
  const h=vals[0];
  return vals.slice(1).filter(r=>r.some(v=>v!=='')).map(r=>{const o={};h.forEach((k,i)=>o[k]=r[i]);return o});
}
function findBy_(tab,key,value){return objects_(tab).find(r=>clean_(r[key])===clean_(value))||null}
function appendObject_(tab,obj){const sh=sheet_(tab),h=sh.getRange(1,1,1,sh.getLastColumn()).getDisplayValues()[0];sh.appendRow(h.map(k=>obj[k]!==undefined?obj[k]:''))}
function upsertBy_(tab,key,value,obj){
  const sh=sheet_(tab),vals=sh.getDataRange().getDisplayValues(),h=vals[0],keyCol=h.indexOf(key);
  let row=-1;for(let i=1;i<vals.length;i++)if(clean_(vals[i][keyCol])===clean_(value)){row=i+1;break}
  const arr=h.map(k=>obj[k]!==undefined?obj[k]:'');
  if(row>0)sh.getRange(row,1,1,h.length).setValues([arr]);else sh.appendRow(arr);
}
function updateField_(tab,key,value,changes){
  const sh=sheet_(tab),vals=sh.getDataRange().getDisplayValues(),h=vals[0],keyCol=h.indexOf(key);
  for(let i=1;i<vals.length;i++)if(clean_(vals[i][keyCol])===clean_(value)){Object.keys(changes).forEach(k=>{const c=h.indexOf(k);if(c>=0)sh.getRange(i+1,c+1).setValue(changes[k])});return}

/**
 * Run this function ONCE from the Apps Script editor after adding portfolio uploads.
 * It requests the Google Drive permission needed to create the Compass Portfolio folders.
 */
function authorizeCompassDrive() {
  const rootName = 'Compass Portfolio';
  const folders = DriveApp.getFoldersByName(rootName);

  if (folders.hasNext()) {
    folders.next();
  } else {
    DriveApp.createFolder(rootName);
  }

  return 'Google Drive access is authorized for Compass.';
}
